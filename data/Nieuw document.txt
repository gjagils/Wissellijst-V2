# CONFIGURATIE
CLIENT_ID = 'f07f3d5b917a4a019b46259ba063a629'
CLIENT_SECRET = '7ad604e14ce842e2881ddbb00493ca04'
REDIRECT_URI = 'http://localhost:8888/callback'
PLAYLIST_ID = '4xSVIQpVLDBsBvwA10xzqj' # De code uit de URL van je playlist
SCOPE = 'playlist-modify-public playlist-modify-private'

# PADEN (voor Synology)
BASE_DIR = '/app/data'
QUEUE_FILE = os.path.join(BASE_DIR, 'volgende_blokje.txt')
CACHE_PATH = os.path.join(BASE_DIR, '.cache')

def rotate_playlist():
    # Inloggen
    auth_manager = SpotifyOAuth(
        client_id=CLIENT_ID, client_secret=CLIENT_SECRET,
        redirect_uri=REDIRECT_URI, scope=SCOPE, cache_path=CACHE_PATH
    )
    sp = spotipy.Spotify(auth_manager=auth_manager)

    # 1. Lees nieuwe nummers uit bestand
    if not os.path.exists(QUEUE_FILE):
        print("Fout: volgende_blokje.txt niet gevonden!")
        return

    with open(QUEUE_FILE, 'r') as f:
        new_tracks = [line.strip() for line in f.readlines() if line.strip()]

    if len(new_tracks) != 5:
        print(f"Fout: Ik heb {len(new_tracks)} nummers gevonden, maar het moeten er exact 5 zijn!")
        return

    # 2. Haal huidige playlist op
    current_tracks = sp.playlist_items(PLAYLIST_ID, limit=50)['items']
    
    # 3. Verwijder de eerste 5 (de oudste)
    track_ids_to_remove = [item['track']['uri'] for item in current_tracks[:5]]
    sp.playlist_remove_all_occurrences_of_items(PLAYLIST_ID, track_ids_to_remove)
    print(f"Verwijderd: {len(track_ids_to_remove)} nummers.")

    # 4. Voeg de 5 nieuwe toe aan het einde
    sp.playlist_add_items(PLAYLIST_ID, new_tracks)
    print("Nieuwe blokje van 5 toegevoegd aan het einde!")

    # 5. Optioneel: Maak het bestand leeg zodat je niet per ongeluk dubbel post
    # os.remove(QUEUE_FILE) 

if __name__ == "__main__":
    rotate_playlist()
